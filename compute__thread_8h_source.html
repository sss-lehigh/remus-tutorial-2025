112 uintptr_t try_allocate_global(
113 std::size_t size, std::function<uint64_t(uint64_t, uint64_t)> seg_locator,
114 std::function<std::atomic<uint64_t> &(uint64_t, uint64_t)> hint_locator,
115 std::function<uint64_t(rdma_ptr<uint64_t>, uint64_t)> faa,
116 std::function<void(rdma_ptr<uint64_t>, uint64_t)> writer) {
117 while (true) {
118 // Get a MemoryNode and Segment on which to try to allocate
119 //
120 // NB: On subsequent iterations of the loop, if these values do not
121 // change, we'll have an infinite loop.
122 auto [mn_id, seg_id] = mn_alloc_pol_.get_mn_seg();
123 auto base = seg_locator(mn_id, seg_id);
124 // Since our bump allocator doesn't coalesce, if this machine has ever
125 // seen its counter exceed what we need for this alloc to work, then try
126 // to get another mn_id/seg_id.
127 auto &hint = hint_locator(mn_id, seg_id);
128 if (hint + size > seg_size_) {
129 continue;
130 }
131 // Try to FAA to allocate
132 //
133 // NB: The above check was just a hint, so failure is still possible
134 auto bump_counter = rdma_ptr<uint64_t>(
135 base + offsetof(internal::ControlBlock, allocated_));
136 auto offset = faa(bump_counter, size);
137 if (offset + size > seg_size_) {
138 // NB: Due to concurrency, there's no decrementing to try to recover!
139 continue;
140 }
141 // Update the hint unless someone else's subsequent update finishes first
142 uint64_t curr_hint = hint;
143 uint64_t new_hint = offset + size;
144 do {
145 } while ((curr_hint <= new_hint) &&
146 !hint.compare_exchange_strong(curr_hint, new_hint));
147 // This is a fresh allocation, so set the size and zero the padding
148 uint64_t ptr = base + offset;
149 writer(rdma_ptr<uint64_t>(ptr + offsetof(header_t, size_)), size);
150 writer(rdma_ptr<uint64_t>(ptr + offsetof(header_t, padding_)), 0);
151 return ptr + HEADER_SIZE;
152 }
153 REMUS_FATAL("Out of memory"); // This is actually unreachable
154 }

163 std::optional<uintptr_t> try_allocate_local(std::size_t size) {
164 // If we can satisfy it via the freelist, do so, and we're done
167 if (size > ALLOC_MED_THRESH) [[unlikely]] {
168 // For large allocations (>8192), use best-fit allocation from the
169 // best_fit_freelist Find the first chunk in the freelist that can fit
170 // this allocation
171 //
172 // TODO: This is first-fit, not best-fit. That's probably unwise?
173 // Also, reverse iteration is probably smarter.
174 auto it = std::find_if(
175 free_blocks_.begin(), free_blocks_.end(),
176 [size](const auto &chunk) { return chunk.first >= size; });
177 if (it != free_blocks_.end()) {
178 uint64_t ptr = it->second;
179 // TODO: Erasing from a vector is costly. Consider using a deque? Or
180 // does reverse iteration reduce the risk?
181 free_blocks_.erase(it);
182 return ptr + HEADER_SIZE;
183 }
184 } else {
185 auto &freelist = freelists_[size];
186 if (!freelist.empty()) {
187 auto ptr = freelist.back();
188 freelist.pop_back();
189 return ptr + HEADER_SIZE;
190 }
191 }
192 return {};
193 }
