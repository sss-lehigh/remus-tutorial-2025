52 remus::Status SendMessage(T &&msg, Segment &seg, ibv_mr *mr) {
53 static_assert(std::ranges::sized_range<T>);
54
55 auto msg_ptr = std::ranges::data(std::forward<T>(msg));
56 auto msg_elements = std::ranges::size(std::forward<T>(msg)); // elements
57 auto msg_size = msg_elements * sizeof(decltype(*msg_ptr)); // bytes
58
59 // Copy the data into the send segment
60 //
61 // NB: We assume that the receiver knows exactly how many bytes are going
62 // to be sent, and has a suitably large buffer prepared to receive
63 // them. We also assume that seg is big enough to hold the message.
64 //
65 // NB: We could ask the programmer to put the message into a Segment, which
66 // would avoid this copy, but we don't, because Send throughput isn't
67 // our priority.
68 std::memcpy(seg.raw(), msg_ptr, msg_size);
69
70 // Describe the command we're going to send to the RNIC
71 ibv_sge sge; // NB: we don't zero it because we set all its fields below
72 sge.addr = reinterpret_cast<uint64_t>(seg.raw());
73 sge.length = msg_size;
74 sge.lkey = mr->lkey;
75
76 // TODO: Could we use rdma_post_send() instead of ibv_post_send()?
77 //
78 // TODO: Document this bit of code a bit better?
79 ibv_send_wr wr;
80 std::memset(&wr, 0, sizeof(wr));
81 wr.send_flags = IBV_SEND_SIGNALED;
82 wr.num_sge = 1;
83 wr.sg_list = &sge;
84 wr.opcode = IBV_WR_SEND;
85 wr.wr_id = 1; // NB: Unique IDs aren't needed for synchronous sends
86 ibv_send_wr *bad_wr;
87 if (ibv_post_send(id_->qp, &wr, &bad_wr) != 0) {
88 // TODO: Should this be fatal?
89 remus::Status err = {remus::InternalError, ""};
90 err << "ibv_post_send(): " << strerror(errno);
91 return err;
92 }
93
94 // NB: This assumes that the CQ associated with the SQ is synchronous.
95 //
96 // TODO: Document this
97 ibv_wc wc;
98 int comps = rdma_get_send_comp(id_, &wc);
99 while (comps < 0 && errno == EAGAIN) {
100 comps = rdma_get_send_comp(id_, &wc);
101 }
102 // TODO: Are these errors recoverable, or should we panic?
103 if (comps < 0) {
104 // TODO: Is operator << really better than just using std::format here?
105 remus::Status e = {remus::InternalError, {}};
106 return e << "rdma_get_send_comp: {}" << strerror(errno);
107 } else if (wc.status != IBV_WC_SUCCESS) {
108 remus::Status e = {remus::InternalError, {}};
109 return e << "rdma_get_send_comp(): " << ibv_wc_status_str(wc.status);
110 }
111 return remus::Status::Ok();
112 }

121 remus::StatusVal<std::vector<uint8_t>> TryDeliverMessage(Segment &seg) {
122 ibv_wc wc;
123 auto ret = rdma_get_recv_comp(id_, &wc);
124 if (ret < 0 && errno != EAGAIN) {
125 remus::Status e = {remus::InternalError, {}};
126 e << "rdma_get_recv_comp: " << strerror(errno);
127 return {e, {}};
128 } else if (ret < 0 && errno == EAGAIN) {
129 return {{remus::Unavailable, "Retry"}, {}};
130 } else {
131 switch (wc.status) {
132 case IBV_WC_WR_FLUSH_ERR:
133 return {{remus::Aborted, "QP in error state"}, {}};
134 case IBV_WC_SUCCESS: {
135 // Prepare the response.
136 std::span<uint8_t> recv_span{seg.raw(), wc.byte_len};
137 remus::StatusVal<std::vector<uint8_t>> res = {
138 remus::Status::Ok(), std::make_optional(std::vector<uint8_t>(
139 recv_span.begin(), recv_span.end()))};
140 return res;
141 }
142 default: {
143 remus::Status err = {remus::InternalError, {}};
144 err << "rdma_get_recv_comp(): " << ibv_wc_status_str(wc.status);
145 return {err, {}};
146 }
147 }
148 }

215 ~Connection() {
216 // A loopback connection is made manually, so we do not need to deal with
217 // the regular `rdma_cm` handling. Similarly, we avoid destroying the event
218 // channel below since it is destroyed along with the id.
219 if (!is_loopback_) {
220 rdma_disconnect(id_);
221 rdma_cm_event *event;
222 auto result = rdma_get_cm_event(id_->channel, &event);
223 while (result == 0) {
224 RDMA_CM_ASSERT(rdma_ack_cm_event, event);
225 result = rdma_get_cm_event(id_->channel, &event);
226 }
227 }
228
229 // We only allocate contexts for connections that were created by the
230 // `RdmaReceiver` callbacks. Otherwise, we created an event channel so
231 // that we could asynchronously connect (and avoid distributed deadlock).
232 auto *context = id_->context;
233 auto *channel = id_->channel;
234 rdma_destroy_ep(id_);
235 if (context != nullptr)
236 free(context);
237 else if (!is_loopback_)
238 rdma_destroy_event_channel(channel);
239 }
