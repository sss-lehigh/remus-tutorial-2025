100 void handle_connections() {
101 REMUS_INFO("MemoryNode {} listening on {}:{}", self_.id, address_, port_);
102 while (remaining_conns_ > 0) {
103 // Attempt to read an event from `listen_channel_`
104 rdma_cm_event *event = nullptr;
105 // Handle errors. Hopefully it's EAGAIN so we can retry
106 if (int ret = rdma_get_cm_event(listen_channel_, &event); ret != 0) {
107 if (errno != EAGAIN) {
108 REMUS_FATAL("rdma_get_cm_event(): {}", strerror(errno));
109 } else {
110 // TODO: Why yield? Is there harm in trying again immediately? If
111 // so, then why not sleep?
112 std::this_thread::yield();
113 continue;
114 }
115 }
116
117 // Handle whatever event we just received
118 rdma_cm_id *id = event->id;
119 switch (event->event) {
120 case RDMA_CM_EVENT_TIMEWAIT_EXIT:
121 // We aren't currently getting CM_EVENT_TIMEWAIT_EXIT events, but if
122 // we did, we'd probably just ACK them and continue
123 rdma_ack_cm_event(event);
124 break;
125
126 case RDMA_CM_EVENT_CONNECT_REQUEST:
127 // This is the main thing we expect: a request for a new connection
128 on_connect(id, event, listen_id_->pd);
129 break;
130
131 case RDMA_CM_EVENT_ESTABLISHED:
132 // Once the connection is fully established, we just ack it, and then
133 // it's ready for use.
134 rdma_ack_cm_event(event);
135 break;
136
137 case RDMA_CM_EVENT_DISCONNECTED:
138 // Since we're polling on a *listening* channel, we're never going to
139 // see disconnected events. If we did, we have some code for knowing
140 // what to do with them.
141 rdma_ack_cm_event(event);
142 on_disconnect(id);
143 break;
144
145 case RDMA_CM_EVENT_DEVICE_REMOVAL:
146 // We don't expect to ever see a device removal event
147 REMUS_FATAL("event: {}, error: {}\n", rdma_event_str(event->event),
148 event->status);
149 break;
150
151 case RDMA_CM_EVENT_ADDR_ERROR:
152 case RDMA_CM_EVENT_ROUTE_ERROR:
153 case RDMA_CM_EVENT_UNREACHABLE:
154 case RDMA_CM_EVENT_ADDR_RESOLVED:
155 case RDMA_CM_EVENT_REJECTED:
156 case RDMA_CM_EVENT_CONNECT_ERROR:
157 // These signals are sent to a connecting endpoint, so we should not
158 // see them here.~ If they appear, abort.
159 REMUS_FATAL("Unexpected signal: {}", rdma_event_str(event->event));
160
161 default:
162 // We did not design for other events, so crash if another event
163 // arrives
164 REMUS_FATAL("Not implemented: {}", rdma_event_str(event->event));
165 }
166 }
167 }

170 void on_connect(rdma_cm_id *id, rdma_cm_event *event, ibv_pd *pd) {
171 // The private data is used to figure out the node that made the request
172 REMUS_ASSERT(event->param.conn.private_data != nullptr,
173 "Received connect request without private data.");
174 uint32_t machine_id = *(uint32_t *)(event->param.conn.private_data);
175 if (machine_id == self_.id) {
176 REMUS_FATAL("on_connect called for self-connection");
177 }
178 // Set the QP ACK timeout before creating the QP
179 uint8_t timeout = 12; // Example timeout value
180 int ret = rdma_set_option(id, RDMA_OPTION_ID, RDMA_OPTION_ID_ACK_TIMEOUT,
181 &timeout, sizeof(timeout));
182 REMUS_ASSERT(ret == 0, "rdma_set_option(): {}", strerror(errno));
183
184 // Create a new QP for the connection.
185 ibv_qp_init_attr init_attr = internal::make_default_qp_init_attrs();
186 ret = rdma_create_qp(id, pd, &init_attr);
187 REMUS_ASSERT(ret == 0, "rdma_create_qp(): {}", strerror(errno));
188
189 // Prepare the necessary resources for this connection. Includes a QP and
190 // memory for 2-sided communication. The underlying QP is RC, so we reuse
191 // it for issuing 1-sided RDMA too. We also store the `machine_id`
192 // associated with this id so that we can reference it later.
193 //
194 // TODO: This config should be double-checked
195 auto context = new IdContext{machine_id, {}};
196 context->conn_param_.private_data = &context->machine_id_;
197 context->conn_param_.private_data_len = sizeof(context->machine_id_);
198 context->conn_param_.rnr_retry_count = 7; // Retry forever
199 context->conn_param_.retry_count = 255;
200 context->conn_param_.responder_resources = 255;
201 context->conn_param_.initiator_depth = 255;
202 id->context = context;
203 internal::make_nonblocking(id->recv_cq->channel->fd);
204 internal::make_nonblocking(id->send_cq->channel->fd);
205
206 // Save the connection and ack it
207 auto conn = new internal::Connection(self_.id, machine_id, id);
208 conns_.emplace_back(conn);
209
210 ret = rdma_accept(id,
211 machine_id == self_.id ? nullptr : &context->conn_param_);
212 REMUS_ASSERT(ret == 0, "rdma_accept(): {}",
213 strerror((*__errno_location())));
214
215 // TODO: We're not checking for errors on this ack?!?
216 rdma_ack_cm_event(event);
217 std::this_thread::sleep_for(std::chrono::milliseconds(100));
218 // TODO: Should Send have fail-stop semantics instead of returning a status
219 // that needs to be checked?
220 auto status = conn->Send(ris_, send_seg_, mr_.get());
221 if (status.t != remus::Ok) {
222 REMUS_FATAL("error in mem node: {}", status.message.value());
223 }
224 remaining_conns_--;
225 }

262 explicit MemoryNode(const MachineInfo &self,
263 std::shared_ptr<remus::ArgMap> args)
264 : self_(self),
265 conns_(),
266 send_seg_(1 << 20),
267 total_threads_(args->uget(remus::CN_THREADS) *
268 (args->uget(remus::LAST_CN_ID) -
269 args->uget(remus::FIRST_CN_ID) + 1)) {
270 int id = args->uget(remus::NODE_ID);
271 uint64_t num_segs = args->uget(remus::SEGS_PER_MN);
272 uint64_t seg_size_bits = args->uget(remus::SEG_SIZE);
273 REMUS_INFO("Node {}: Configuring Memory Node ({} segments at 2^{}B each)",
274 id, num_segs, seg_size_bits);
275
276 // How many non-localhost compute nodes are there?
277 int c0 = args->uget(remus::FIRST_CN_ID);
278 int cn = args->uget(remus::LAST_CN_ID);
279 uint32_t cns = cn - c0 + 1;
280 if (id >= c0 && id <= cn) cns--;
281 uint32_t qlw = args->uget(remus::QP_LANES);
282 remaining_conns_ = cns * qlw;
283
284 // Make the listening endpoint, and register the sending segment with it
285 uint16_t port = args->uget(remus::MN_PORT);
286 listen_id_ = remus::internal::make_listen_id(self.address, port);
287 REMUS_ASSERT(listen_id_->pd != nullptr, "Error creating protection domain");
288 mr_ = send_seg_.registerWithPd(listen_id_->pd);
289
290 // Construct the memory pools, configure their control region, and register
291 // them with the listening endpoint
292 for (uint64_t i = 0; i < num_segs; ++i) {
293 auto seg = std::make_unique<internal::Segment>(1ULL << seg_size_bits);
294 new ((internal::ControlBlock *)(seg->raw()))
295 internal::ControlBlock(1ULL << seg_size_bits);
296 auto mr = seg->registerWithPd(listen_id_->pd);
297 segs_.emplace_back(SegInfo{std::move(seg), std::move(mr)});
298 }
299
300 // Prepare the RegionInfo for this node
301 //
302 // NB: We could be more efficient if we constructed the ris in a segment,
303 // but we're not concerned about the performance of Send(), since we
304 // only use it for network set-up.
305 for (uint64_t i = 0; i < num_segs; ++i) {
306 ris_.emplace_back((uintptr_t)segs_[i].seg_->raw(), segs_[i].mr_->rkey);
307 }
308 REMUS_INFO("Shared Segments:");
309 for (auto ri : ris_) {
310 REMUS_INFO(" 0x{:x} (rk=0x{:x})", ri.raddr, ri.rkey);
311 }
312
313 // Initialize a listening endpoint, and then park a thread on it, so the
314 // thread can receive new connection requests
315 //
316 // NB: Since we are using an async socket, we call `listen()` here, but it
317 // won't block or run any code for handling a listening request. We'll
318 // do that later, in a separate thread.
319 REMUS_INFO("Listener thread awaiting {} connections", remaining_conns_);
320 listen_channel_ = rdma_create_event_channel();
321 if (rdma_migrate_id(listen_id_, listen_channel_) != 0) {
322 REMUS_FATAL("rdma_migrate_id(): {}", strerror(errno));
323 }
324 internal::make_nonblocking(listen_id_->channel->fd);
325 if (rdma_listen(listen_id_, 0) != 0) {
326 REMUS_FATAL("rdma_listen(): {}", strerror(errno));
327 }
328 address_ = std::string(inet_ntoa(
329 reinterpret_cast<sockaddr_in *>(rdma_get_local_addr(listen_id_))
330 ->sin_addr));
331 port_ = rdma_get_src_port(listen_id_);
332 runner_ = std::thread([&]() { handle_connections(); });
333 }
