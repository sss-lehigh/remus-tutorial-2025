156 void import(const std::initializer_list<Arg> &in) {
157 using namespace std;
158 if (program_name != "") {
159 throw runtime_error("Error: cannot call import_args() after parse()");
160 }
161 for (auto c : in) {
162 if ((c.flag.length() < 2) || (*c.flag.begin() != '-')) {
163 throw runtime_error(format("Error: invalid flag `{}`", c.flag));
164 }
165 if (args.find(c.flag) != args.end()) {
166 throw runtime_error(format("Error: duplicate flag `{}`", c.flag));
167 }

169 }

179 void parse(int argc, char **argv) {
180 using namespace std;
181 // Ensure this wasn't called more than once
182 if (program_name != "") {
183 throw runtime_error("Error: parse() should only be called once!");
184 }
185 program_name = basename(argv[0]);
186
187 int curr = 1;
188 while (curr < argc) {
189 string ca = argv[curr];
190 auto arg = args.find(ca);
191 if (arg == args.end()) {
192 usage();
193 throw runtime_error(format("Error: unrecognized argument `{}`", ca));
194 }
195 // Handle bools first, because they don't take a value
196 if (arg->second.type == Arg::BOOL) {
197 arg->second.value = true;
198 ++curr;
199 continue;
200 }
201 // Fail if we're at the end, and there's no value
202 if (curr == (argc - 1)) {
203 usage();
204 throw runtime_error(
205 format("Error: argument `{}` requires a value", ca));
206 }
207 // Fail if the next thing isn't a value
208 string next = argv[curr + 1];
209 if (*next.begin() == '-') {
210 usage();
211 throw runtime_error(
212 format("Error: argument `{}` requires a value", ca));
213 }
214
215 // Now we can parse the value and advance
216 if (arg->second.type == Arg::U64) {
217 arg->second.value = stoull(next);
218 } else if (arg->second.type == Arg::F64) {
219 arg->second.value = stod(next);
220 } else if (arg->second.type == Arg::STR) {
221 arg->second.value = next;
222 if (arg->second.options.size() > 0) {
223 bool match = false;
224 for (auto i : arg->second.options) {
225 match = match || (i == next);
226 }
227 if (!match) {
228 usage();
229 throw runtime_error(
230 format("Error: invalid value for argument `{}`", ca));
231 }
232 }
233 }
234 // TODO: I don't see where we handle Bool args?
235 curr += 2;
236 }
237
238 // Verify that no required args were skipped
239 for (auto a : args) {
240 if (!a.second.value) {
241 usage();
242 throw runtime_error(format("Error: `{}` is required", a.second.flag));
243 }
244 }
245 }
