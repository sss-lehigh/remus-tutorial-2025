66inline std::vector<int> find_active_ports(ibv_context *context) {
67 // Find the first active port, failing if none exists.
68 //
69 // NB: port 0 is a control port, so we start at 1
70 ibv_device_attr dev_attr;
71 ibv_query_device(context, &dev_attr);
72 std::vector<int> ports;
73 for (int i = 1; i <= dev_attr.phys_port_cnt; ++i) {
74 ibv_port_attr port_attr;
75 ibv_query_port(context, i, &port_attr);
76 if (port_attr.state == IBV_PORT_ACTIVE) {
77 ports.push_back(i);
78 }
79 }
80 return ports;
81}

85inline std::vector<std::pair<std::string, int>> get_avail_devices() {
86 int num_devices;
87 auto **device_list = ibv_get_device_list(&num_devices);
88 if (num_devices <= 0) return {};
89 std::vector<std::pair<std::string, int>> active;
90 for (int i = 0; i < num_devices; ++i) {
91 if (auto *context = ibv_open_device(device_list[i]); context) {
92 for (auto p : find_active_ports(context)) {
93 active.emplace_back(context->device->name, p);
94 }
95 }
96 }
97 ibv_free_device_list(device_list);
98 return active;
99}

118inline rdma_cm_id *make_listen_id(const std::string &address, uint16_t port) {
119 rdma_cm_id *listen_id;
120 // Check that devices exist before trying to set things up.
121 auto devices = get_avail_devices();
122 if (devices.empty()) {
123 REMUS_FATAL("CreateListeningEndpoint :: no RDMA-capable devices found");
124 }
125
126 // Get the local connection information.
127 rdma_addrinfo hints, *resolved;
128 std::memset(&hints, 0, sizeof(rdma_addrinfo));
129 hints.ai_flags = RAI_PASSIVE;
130 hints.ai_port_space = RDMA_PS_TCP;
131 auto port_str = std::to_string(htons(port));
132 int gai_ret =
133 rdma_getaddrinfo(address.c_str(), port_str.c_str(), &hints, &resolved);
134 if (gai_ret != 0) {
135 REMUS_FATAL("rdma_getaddrinfo(): {}", gai_strerror(gai_ret));
136 }
137
138 REMUS_ASSERT(resolved != nullptr, "Did not find an appropriate RNIC");
139
140 // Create an endpoint to receive incoming requests
141 ibv_qp_init_attr init_attr;
142 std::memset(&init_attr, 0, sizeof(ibv_qp_init_attr));
143 init_attr.cap.max_send_wr = init_attr.cap.max_recv_wr = kMaxWr;
144 init_attr.cap.max_send_sge = init_attr.cap.max_recv_sge = 1;
145 init_attr.cap.max_inline_data = 0;
146 init_attr.sq_sig_all = 1;
147 auto err = rdma_create_ep(&listen_id, resolved, nullptr, &init_attr);
148 rdma_freeaddrinfo(resolved);
149 if (err != 0) {
150 REMUS_FATAL("listener rdma_create_ep():{} for {}:{}", strerror(errno),
151 address, port);
152 }
153 return listen_id;
154}
