42inline rdma_cm_id *initialize_ep(std::string_view address, uint16_t port) {
43 // Compute the info for the node we're connecting to
44 auto port_str = std::to_string(htons(port));
45 rdma_addrinfo hints, *resolved = nullptr;
46 std::memset(&hints, 0, sizeof(hints));
47 struct sockaddr_in src;
48 std::memset(&src, 0, sizeof(src));
49 hints.ai_port_space = RDMA_PS_TCP;
50 hints.ai_qp_type = IBV_QPT_RC;
51 hints.ai_family = AF_IB;
52 hints.ai_src_len = sizeof(src);
53 src.sin_family = AF_INET;
54 // inet_aton(address_.data(), &src.sin_addr);
55 hints.ai_src_addr = reinterpret_cast<sockaddr *>(&src);
56 if (int err =
57 rdma_getaddrinfo(address.data(), port_str.data(), &hints, &resolved);
58 err != 0) {
59 REMUS_FATAL("rdma_getaddrinfo(): {}", gai_strerror(err));
60 }
61
62 // Start making a connection
63 ibv_qp_init_attr init_attr = make_default_qp_init_attrs();
64 rdma_cm_id *id = nullptr;
65 auto err = rdma_create_ep(&id, resolved, nullptr, &init_attr);
66 rdma_freeaddrinfo(resolved);
67 if (err) {
68 REMUS_FATAL("compute node rdma_create_ep(): {}", strerror(errno));
69 }
70 return id;
71}

84inline Connection *connect_remote(uint32_t my_id, uint32_t mn_id,
85 std::string_view mn_addr, uint16_t port,
86 internal::Segment &seg,
87 std::vector<internal::ibv_mr_ptr> &mrs) {
88 uint32_t backoff_us_ = 0;
89 while (true) {
90 // TODO: Need a one-line comment here explaining this block of code
91 rdma_cm_id *id = initialize_ep(mn_addr, port);
92 auto mr = seg.registerWithPd(id->pd);
93 RDMA_CM_ASSERT(rdma_post_recv, id, nullptr, seg.raw(), seg.capacity(),
94 mr.get());
95 mrs.push_back(std::move(mr));
96
97 // Migrate the new endpoint to a nonblocking event channel and do more
98 // config
99 auto *event_channel = rdma_create_event_channel();
100 make_nonblocking(event_channel->fd);
101 if (rdma_migrate_id(id, event_channel) != 0) {
102 REMUS_FATAL("rdma_migrate_id(): {}", strerror(errno));
103 }
104
105 // Set the QP ACK timeout
106 uint8_t timeout = 12; // Example timeout value
107 if (rdma_set_option(id, RDMA_OPTION_ID, RDMA_OPTION_ID_ACK_TIMEOUT,
108 &timeout, sizeof(timeout)) != 0) {
109 REMUS_FATAL("rdma_set_option(): {}", strerror(errno));
110 }
111
112 // TODO: These values feel like they need more documentation
113 rdma_conn_param conn_param;
114 std::memset(&conn_param, 0, sizeof(conn_param));
115 conn_param.private_data = &my_id;
116 conn_param.private_data_len = sizeof(my_id);
117 conn_param.retry_count = 255;
118 conn_param.rnr_retry_count = 7;
119 conn_param.responder_resources = 8;
120 conn_param.initiator_depth = 8;
121 if (rdma_connect(id, &conn_param) != 0) {
122 REMUS_FATAL("rdma_connect(): {}", strerror(errno));
123 }
124
125 // It takes a few events before the channel is ready to use, so we need an
126 // event loop here, to receive events and move through a set of transitions
127 while (true) {
128 // Poll until we get an event
129 rdma_cm_event *event;
130 auto result = rdma_get_cm_event(id->channel, &event);
131 while (result < 0 && errno == EAGAIN) {
132 result = rdma_get_cm_event(id->channel, &event);
133 }
134
135 // Save the event and ack it
136 auto cm_event = event->event;
137 if (rdma_ack_cm_event(event) != 0) {
138 REMUS_FATAL("rdma_ack_cm_event(): {}", strerror(errno));
139 }
140
141 // On an "established" event, we can make and save the connection
142 if (cm_event == RDMA_CM_EVENT_ESTABLISHED) {
143 make_sync(event_channel->fd);
144 make_nonblocking(id->recv_cq->channel->fd);
145 make_nonblocking(id->send_cq->channel->fd);
146
147 // Make and return the connection
148 return new Connection(my_id, mn_id, id);
149 }
150
151 // On an ADDR_RESOLVED, we just ack (which we did above)
152 else if (cm_event == RDMA_CM_EVENT_ADDR_RESOLVED) {
153 }
154
155 // If we get a REJECTED, cleanup, backoff, try it all again
156 else if (cm_event == RDMA_CM_EVENT_REJECTED) {
157 rdma_destroy_ep(id);
158 rdma_destroy_event_channel(event_channel);
159 backoff_us_ = backoff_us_ > 0
160 ? std::min((backoff_us_ + (100 * my_id)) * 2,
161 connect_backoff_max_us)
162 : connect_backoff_min_us;
163 std::this_thread::sleep_for(std::chrono::microseconds(backoff_us_));
164 break;
165 }
166
167 // Otherwise fail
168 else {
169 REMUS_FATAL("Got unexpected event: {}", rdma_event_str(cm_event));
170 }
171 }
172 }

183inline Connection *connect_loopback(uint32_t my_id, std::string_view address,
184 uint16_t port) {
185 // Do the initial endpoint configuration
186 rdma_cm_id *id = initialize_ep(address, port);
187
188 // Query the ports to find one that is available and appropriate
189 ibv_device_attr dev_attr;
190 if (ibv_query_device(id->verbs, &dev_attr) != 0) {
191 REMUS_FATAL("ibv_query_device(): {}", strerror(errno));
192 }
193
194 // There are a whole bunch of funny rules and configuration requirements when
195 // connecting via loopback, especially if it's not an IB device.
196 //
197 // TODO: Do we want to drop the non-IB code paths here, since they're
198 // untested? Or maybe factor them out into a helper method that can be
199 // documented with a big "UNTESTED" warning?
200 ibv_port_attr port_attr;
201 uint32_t LOOPBACK_PORT_NUM = 1;
202
203 // use first port that is active for loopback
204 for (int i = 1; i <= dev_attr.phys_port_cnt; ++i) {
205 if (ibv_query_port(id->verbs, i, &port_attr) != 0) {
206 REMUS_FATAL("ibv_query_port(): {}", strerror(errno));
207 }
208 if (port_attr.state == IBV_PORT_ACTIVE) {
209 LOOPBACK_PORT_NUM = i;
210 // REMUS_DEBUG("Using physical port {} for loopback", i);
211 break;
212 }
213 }
214
215 // TODO: These values feel like they need more documentation. Also, we're
216 // configuring loopback differently (via ibv_qp_atttr) than we
217 // configure remote (via rdma_conn_param). Need to verify that the two
218 // techniques result in the same behaviors and performance. I'm
219 // especially confused because we don't do the RoCE or GRH stuff for
220 // Remote connections... why not?
221 ibv_qp_attr attr;
222 std::memset(&attr, 0, sizeof(attr));
223 attr.qp_access_flags = IBV_ACCESS_LOCAL_WRITE | IBV_ACCESS_REMOTE_READ |
224 IBV_ACCESS_REMOTE_WRITE | IBV_ACCESS_REMOTE_ATOMIC;
225 attr.max_dest_rd_atomic = 8;
226 attr.path_mtu = IBV_MTU_4096;
227 attr.min_rnr_timer = 12;
228 attr.rq_psn = 0;
229 attr.sq_psn = 0;
230 attr.timeout = 12;
231 attr.retry_cnt = 255;
232 attr.rnr_retry = 7;
233 attr.max_rd_atomic = 8;
234 attr.qp_state = IBV_QPS_INIT;
235 attr.port_num = LOOPBACK_PORT_NUM;
236 int attr_mask =
237 IBV_QP_STATE | IBV_QP_PKEY_INDEX | IBV_QP_PORT | IBV_QP_ACCESS_FLAGS;
238 if (ibv_modify_qp(id->qp, &attr, attr_mask) != 0) {
239 REMUS_FATAL("ibv_modify_qp(): {}", strerror(errno));
240 }
241 attr.ah_attr.dlid = port_attr.lid;
242 attr.ah_attr.port_num = LOOPBACK_PORT_NUM;
243
244 // TODO: This documentation isn't clear to a non-expert. What are LID, GRH?
245 //
246 // This LID is invalid and likely RoCE, so this is a hack to get around that
247 // or the GRH is required regardless
248 if (port_attr.lid == 0x0 || (port_attr.flags & IBV_QPF_GRH_REQUIRED) != 0) {
249 // Our address handle has a global route
250 attr.ah_attr.is_global = 1;
251
252 // We query the first GID, which should always exist There may be others,
253 // but that shouldn't impact anything for us
254 //
255 // We can go from gid = 0 to gid = port_attr.gid_table_len - 1
256 REMUS_ASSERT(port_attr.gid_tbl_len >= 1,
257 "Need a gid table that has at least one entry");
258 ibv_gid gid;
259 if (ibv_query_gid(id->verbs, LOOPBACK_PORT_NUM, 0, &gid)) {
260 REMUS_FATAL("Fail on query gid");
261 }
262
263 // Set our gid to the gid index we queried
264 attr.ah_attr.grh.dgid = gid;
265 attr.ah_attr.grh.sgid_index = 0;
266 // allow for the max number of hops
267 attr.ah_attr.grh.hop_limit = 0xFF;
268 attr.ah_attr.grh.traffic_class = 0; // some traffic class
269 // non-zero is support to give a hint to switches
270 // but we dont care; this is loopback
271 attr.ah_attr.grh.flow_label = 0;
272 }
273
274 // TODO: Document this chunk of config
275 attr.qp_state = IBV_QPS_RTR;
276 attr.dest_qp_num = id->qp->qp_num;
277 attr_mask =
278 (IBV_QP_STATE | IBV_QP_AV | IBV_QP_PATH_MTU | IBV_QP_DEST_QPN |
279 IBV_QP_RQ_PSN | IBV_QP_MAX_DEST_RD_ATOMIC | IBV_QP_MIN_RNR_TIMER);
280 if (ibv_modify_qp(id->qp, &attr, attr_mask) != 0) {
281 REMUS_FATAL("ibv_modify_qp(): {}", strerror(errno));
282 }
283 attr.qp_state = IBV_QPS_RTS;
284 attr_mask = (IBV_QP_STATE | IBV_QP_SQ_PSN | IBV_QP_TIMEOUT |
285 IBV_QP_RETRY_CNT | IBV_QP_RNR_RETRY | IBV_QP_MAX_QP_RD_ATOMIC);
286 if (ibv_modify_qp(id->qp, &attr, attr_mask) != 0) {
287 REMUS_FATAL("ibv_modify_qp(): {}", strerror(errno));
288 }
289 make_nonblocking(id->recv_cq->channel->fd);
290 make_nonblocking(id->send_cq->channel->fd);
291
292 // Make and return the connection
293 return new Connection(my_id, my_id, id);

394 void save_region(uint16_t node_id, uint64_t region, uint32_t rkey) {
395 constexpr uint64_t node_mask = 0xFFFFULL << 48;
396 REMUS_ASSERT((region & node_mask) == 0, "Top bits of region must be 0");
397 // TODO: Should we also assert that the low bits are 0? We expect
398 // alignment, after all...
399 REMUS_ASSERT((region & seg_mask_) == 0,
400 "Region is not aligned to segment size");
401 uint64_t n = node_id;
402 n = n << 48;
403 uint64_t key = n | region;
404 // Only save it once, because it's per-node, not per-connection
405 if (segment_rkeys_.find(key) == segment_rkeys_.end()) {
406 REMUS_INFO(" Received segment 0x{:x} from node {} with rkey {}", region,
407 node_id, rkey);
408 segment_rkeys_.insert({key, rkey});
409 // save this region to the seg_ map, so we can allocate out of it
410 //
411 // NB: Rather than do an rdma_read, we assume that nothing has been
412 // allocated yet, and thus the bump allocator FAA hint should be the
413 // tail of the segment's metadata.
414 segs_[node_id].emplace_back(key, sizeof(internal::ControlBlock));
415 }
416 }

473 void connect_local(std::vector<MachineInfo> &memnodes,
474 std::vector<internal::RegionInfo> local_rkeys) {
475 uint64_t qp_lanes = args_->uget(remus::QP_LANES);
476 uint32_t port = args_->uget(remus::MN_PORT);
477
478 for (auto &p : memnodes) {
479 if (p.id == self_.id) {
480 for (uint64_t i = 0; i < qp_lanes; ++i) {
481 // Connect, then register the big segment with that connection
482 REMUS_INFO("Connecting to localhost {}:{} (id = {})", p.address, port,
483 p.id);
484 auto conn = internal::connect_loopback(self_.id, self_.address, port);
485 mrs_.push_back(seg_.registerWithPd(conn->pd()));
486
487 // Save the connection and the regions
488 auto lkey = mrs_.back()->lkey;
489 save_conn(p.id, conn, lkey);
490 for (auto &r : local_rkeys) {
491 save_region(p.id, r.raddr, r.rkey);
492 }
493 }
494 }
