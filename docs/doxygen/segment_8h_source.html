48inline std::optional<uintptr_t> find_mmap_location(uintptr_t min_addr,
49 size_t len) {
50 // Ensure non-zero power of two for min_addr and len
51 assert(min_addr && !(min_addr & (min_addr - 1)));
52 if (!(len && !(len & (len - 1)))) {
53 REMUS_FATAL("len is not a power of 2");
54 }
55
56 // Round the initial guess up to the next multiple of len
57 uintptr_t addr = roundup(min_addr, len);
58
59 // Start going through the procfile. We assume each line is <= 2048B
60 //
61 // The format is: `l-u p o d i f`
62 // - l, u: The lower and upper bound of the valid memory region (hex)
63 // - p: The permissions of that region
64 // - o: The offset of the file that is mapped (if it's a mmap'd file)
65 // - d: The (hex) device number if it's a mmap'd file
66 // - i: The inode / file number if it's a mmap'd file
67 // - f: The filename if it's a mmap'd file
68 //
69 // Note that lines are sorted by l, and that entries do not overlap. Thus if
70 // we pull in each line, then use scanf to get l-u, we have all the info we
71 // need to know that a given region is mapped. Everything else is available
72 FILE *fp;
73 if ((fp = fopen("/proc/self/maps", "r")) == NULL) {
74 return {};
75 }
76 char line[2048];
77 while (fgets(line, 2048, fp) != NULL) {
78 uintptr_t l = 0, u = 0; // Lower and upper bounds of the ranges in the file
79 if (sscanf(line, "%lx-%lx", &l, &u) == 2) {
80 if ((addr + len) <= l) {
81 break; // addr is likely to work!
82 }
83 if (addr < u) {
84 addr = roundup(u, len); // Move past u, to next multiple of len
85 }
86 }
87 }
88 fclose(fp);
89
90 // Check for overflow: If we would run off the edge of the address map, fail
91 if ((addr + len) < addr) {
92 return {};
93 }
94 return addr;
95}

157 Segment(uint64_t cap) : capacity_(cap) {
158 // Get aligned memory via mmap
159 auto hint = find_mmap_location((1UL << 35), capacity_);
160 int flags = MAP_PRIVATE | MAP_ANONYMOUS | MAP_FIXED_NOREPLACE;

167 raw_ = (uint8_t *)mmap((void *)hint.value(), capacity_,
