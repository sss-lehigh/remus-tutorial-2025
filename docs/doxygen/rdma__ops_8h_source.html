214template <typename T>
215inline void Poll(Connection *conn, std::atomic<int> *ack, rdma_ptr<T> ptr) {
216 // Poll until we match on the condition
217 ibv_wc wc;
218 while (*ack != 0) {
219 int poll = conn->poll_cq(1, &wc);
220 if (poll == 0 || (poll < 0 && errno == EAGAIN))
221 continue;
222 // TODO: Why doesn't write look for poll != 1, but this does?
223 if (poll != 1 || wc.status != IBV_WC_SUCCESS) {
224 REMUS_ASSERT(poll == 1 && wc.status == IBV_WC_SUCCESS,
225 "ibv_poll_cq(): {} @ {}",
226 (poll < 0 ? strerror(errno) : ibv_wc_status_str(wc.status)),
227 format_rdma_ptr(ptr));
228 }
229 int old = ((std::atomic<int> *)wc.wr_id)->fetch_sub(1);
230 REMUS_ASSERT(old >= 1, "Broken synchronization");
231 }
232}

241template <typename T>
242bool PollAsync(Connection *conn, std::atomic<int> *ack, rdma_ptr<T> ptr) {
243 // Poll until we match on the condition
244 ibv_wc wc;
245 if (*ack != 0) {
246 int poll = conn->poll_cq(1, &wc);
247 if (poll == 0 || (poll < 0 && errno == EAGAIN))
248 return false;
249 // TODO: Why doesn't write look for poll != 1, but this does?
250 if (poll != 1 || wc.status != IBV_WC_SUCCESS) {
251 REMUS_ASSERT(poll == 1 && wc.status == IBV_WC_SUCCESS,
252 "ibv_poll_cq(): {} @ {}",
253 (poll < 0 ? strerror(errno) : ibv_wc_status_str(wc.status)),
254 format_rdma_ptr(ptr));
255 }
256 int old = ((std::atomic<int> *)wc.wr_id)->fetch_sub(1);
257 REMUS_ASSERT(old >= 1, "Broken synchronization");
258 return false;
259 }
260 return true;
261}
